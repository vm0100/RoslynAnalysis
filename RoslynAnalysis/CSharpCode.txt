public class DomainService
{
	/// <summary>
    /// 获取项目下的预约单
    /// </summary>
    /// <param name="projIds">项目GUID</param>
    /// <returns></returns>
    public virtual List<Model.Booking> GetBookingsByProjIds(List<Guid> projIds)
    {
        var obj = new Booking();
        obj.BookName = "";

        return _entityService.Instance.Find(p => p.ProjGUID.In(projIds));
    }
}
